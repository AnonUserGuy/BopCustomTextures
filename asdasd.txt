        Sprite replacement = Sprite.Create(
            original.texture,
            original.rect,
            original.pivot / original.rect.size,
            original.pixelsPerUnit,
            0,
            SpriteMeshType.Tight,
            original.border
        );

        Vector2[] adjustedVertices = new Vector2[original.vertices.Length];
        for (int i = 0; i < original.vertices.Length; i++)
        {
            // Shift from original rect position and scale
            Vector2 v = original.vertices[i];
            v.x = (v.x + originalRect.x) * scaleX;
            v.y = (v.y + originalRect.y) * scaleY;
            adjustedVertices[i] = v;
        }

        // Copy triangles
        replacement.OverrideGeometry(adjustedVertices, original.triangles);

        replacement.name = original.name;
        SpriteMaps[scene][original.name] = replacement;
